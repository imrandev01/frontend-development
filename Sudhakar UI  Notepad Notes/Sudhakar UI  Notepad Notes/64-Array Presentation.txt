Creating and Adding elements into page dynamically:

1. You can create any HTML element by using

        document.createElement()

Syntax:
        document.createElement("img")
        document.createElement("p")
        document.createElement("div")

2. Assign created element to a memory reference

Syntax:
        var pic = document.createElement("img");    [HTMLElement type]

3. Every dynamic element requires "Properties"

Syntax:
        pic.width = "200";
        pic.height= "100";
        pic.src= "./images/name.jpg";

4. Add element into page by using "appendChild()"

Syntax:
        <body>  </body>
        <div id="container"> </div>

        document.querySelector("body").appendChild(pic);
        document.getElementById("container").appendChild(pic);


Ex:
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
    <script>
        function LoadClick(){
            var pic = document.createElement("img");
           
            pic.width="100";
            pic.height="100";
            pic.src="../public/images/women-fashion.jpg";

            document.getElementById("container").appendChild(pic);

        }
    </script>
</head>
<body>
    <button onclick="LoadClick()">Load Image</button>
    <br><br>
    <div id="container">
       
    </div>
</body>
</html>

         var menu = ["Amazon","Home", "Shop", "Blog", "Pages", "Docs"];
         function bodyload(){
              menu.map(function(item){
                  var div = document.createElement("div");
                  div.className = "alert alert-dismissible alert-warning";
                  div.innerHTML = `
                      <button class="btn btn-close" data-bs-dismiss="alert"></button>
                      <h3>${item}</h3>
                  `;
                  document.querySelector("section").appendChild(div);
              })
         }
    </script>
</head>
<body onload="bodyload()" class="container-fluid">
    <section class="m-4 p-4 w-25">

    </section>
</body>
