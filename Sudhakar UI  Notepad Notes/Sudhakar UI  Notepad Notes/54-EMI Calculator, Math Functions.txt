<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>EMI Calculator</title>
    <link rel="stylesheet" href="../node_modules/bootstrap/dist/css/bootstrap.css">
    <style>
        input[type="range"]{
            width:90%;
        }
    </style>
    <script type="text/javascript">
        function AmountChange(){
            document.getElementById("txtAmount").value = document.getElementById("rangeAmount").value;
        }
        function YearChange(){
            document.getElementById("txtYears").value = document.getElementById("rangeYears").value;
        }
        function RateChange(){
            document.getElementById("txtRate").value = document.getElementById("rangeRate").value;
        }
        function CalculateClick(){
            var P = parseInt(document.getElementById("txtAmount").value);
            var r = parseFloat(document.getElementById("txtRate").value)/12/100;
            var n = parseInt(document.getElementById("txtYears").value) * 12;
            var EMI = P * r * (Math.pow(1+r,n)) / Math.pow(1+r,n) - 1;
            document.getElementById("result").innerHTML = "Your installement amount every month is <b>" + Math.round(EMI).toLocaleString('en-in',{style:'currency', currency:'INR'}) + "</b> for " + n + " months";
        }
    </script>
</head>

<!DOCTYPE html>
<html lang="en">

    <title>Document</title>
    <link rel="stylesheet" href="../node_modules/bootstrap-icons/font/bootstrap-icons.css">
    <link rel="stylesheet" href="../node_modules/bootstrap/dist/css/bootstrap.css">
    <script>
        function CalculateClick(){
            var n = parseInt(prompt("Enter Weight"));
            var status = document.getElementById("status");
            if(n<40){
                status.style.marginLeft = "100px";
            } else if (n>40 && n<60) {
                status.style.marginLeft = "250px";
            }
        }
    </script>
</head>
<body class="container-fluid">
    <button onclick="CalculateClick()" class="btn btn-primary mt-4">Calculate</button>
    <div class="mt-4 progress">
        <div class="progress-bar bg-dark me-1" style="width:25%">Under Weight</div>
        <div class="progress-bar bg-success me-1" style="width:25%">Normal Weight</div>
        <div class="progress-bar bg-warning me-1" style="width:25%">Over Weight</div>

    </div>
