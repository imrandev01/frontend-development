11. trim()            : It removes the leading spaces in a string. It trims both start and
                  end.

12. trimEnd()        : It removes leading spaces towards end.

13. trimStart()        : It removes leading space towards start.

Syntax:
    var  otp = "  5678";
    otp.trim();                "5678"

Ex:
<script>
     var OTP = prompt("Enter OTP");

     if(OTP.trim()=="5678")
     {
          document.write("Success..");
     }
     else
     {
          document.write("Invalid OTP");
     }
</script>

14. split()            : It splits the string into multiple strings using a delimiter. [separator]

Syntax:
    var  msg = "Welcome to JavaScript";
    var  result = msg.split(' ');

    result[0]        // Welcome
    result[1]        // to
    result[2]        // JavaScript


15. replace()        : It replaces specified char or word with a new char and word.
                  It can replace only the first occurrence value.

Syntax:
     var  msg = "Hello ! JavaScript";
     msg.replace("JavaScript", "HTML");

16. repeat()        : It creates a copy of string. It uses a number that specified the
                  total count of copies to create.

Syntax:
    var msg = "Welcome";
    msg.repeat(2);

               if(password.match(regExp))
               {
